def a(b):
    c = ""
    for d in b:
        if d.isalpha():
            if d.islower():
                c += chr((ord(d) - 97 + 5) % 26 + 97)
            else:
                c += chr((ord(d) - 65 + 5) % 26 + 65)
        else:
            c += d
    return c

x = input("Enter the Password: ")
y = a(x)
if y == "r3whjs4wd":
    print("You WIN!!!!")
else:
    print("WRONG!!!!")
